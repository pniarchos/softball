create table games (
  id uuid primary key default gen_random_uuid(),
  created_at timestamptz default now(),
  opponent text, date text, innings int default 7,
  away_name text default 'CATO', home_name text,
  away_runs int[] default array_fill(0,array[7]),
  home_runs int[] default array_fill(0,array[7]),
  away_r int default 0, away_h int default 0, away_e int default 0,
  home_r int default 0, home_h int default 0, home_e int default 0,
  current_inning int default 1, inning_half text default 'top',
  balls int default 0, strikes int default 0, outs int default 0,
  bases int[] default '{0,0,0}',
  batting_order text[] default '{}',
  status text default 'active'
);
alter table games enable row level security;
create policy "public access" on games for all using (true);
